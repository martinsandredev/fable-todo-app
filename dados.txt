let myInput = document.querySelector("#input-search") :?> Browser.Types.HTMLInputElement

let myButton = document.querySelector("#button-search") :?> Browser.Types.HTMLButtonElement

let myResult = document.querySelector("result-search") :?> Browser.Types.HTMLDivElement

// Register our listener

    searchValue <- myInput.value
    myResult.innerHTML <- searchValue

myInput.oninput <- fun _ ->
    searchValue <- myInput.value
    myResult.innerHTML <- searchValue
